    class Circle():

    def __init__(self, supplied_r):
        self.r = supplied_r

    def area(self):
        A = np.pi*self.r**2
        return A

    def circumference(self):
        C = 2*np.pi*self.r
        return C

    class Newton():

        self.G = value_of_G
        self.info = supplied_info

    Class Dice

    • Lets create a Class called Dice for fun
    class Dice(object):

    def __init__(self,A_value,B_value,C_value):
        self.pi = 3.14
        self.A = A_value
        self.B = B_value
        self.C = C_value 

    def find_sum(self,n1,n2):
        S = n1+n2
        return S

    def find_product(self,n1,n2):
        P = n1*n2
        return P

    def poly(self,x):
        pl = self.A*self.find_product(x,x) + self.B*x + self.C
        return pl

    def roll_dice(self):
        side = random.choice([1,2,3,4,5,6])
        return side

    def roll_two_dices(self):
        d1 = self.roll_dice()
        d2 = self.roll_dice()
        p = self.find_product(d1,d2)
        s = self.find_sum(d1,d2)
        return d1,d2,p,s
